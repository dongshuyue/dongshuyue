4. 柱状图或折线的文本标签

使用series[i]里的label, label拥有 normal(一般状态), emphasis (悬浮状态)两个状态

 
xxxxxxxxxx
option.series[0]={
    name:'系列1',
    type:'bar',
    barWidth: 30,
    data:[10, 52, 200, 334, 390, 330, 220],
    label: {
      normal: {
        show: true, //是否显示， 默认为false
        position: 'top', //文本位置, 绝对的像素值position: [10, 10],相对的百分比position: ['50%', '50%']
        //文本位置，top','left','right','bottom','inside','insideLeft','insideRight','insideTop'
        //'insideBottom','insideTopLeft','insideBottomLeft',insideTopRight','insideBottomRight
        formatter : function (params) {}  //自定义返回文字，不写时从data取值
        textStyle: {//文本样式设置
          fontSize: 14,
          fontColor: '#333',

      }
    }
  }

9.x轴或y轴的文字过长时换行

遇到空格时换行

 
xxxxxxxxxx
option.xAxis = {
   type : 'category',
   data : ['Mon 星期一', 'Tue 星期二', 'Wed 星期三', 'Thu 星期四', 'Fri 星期五', 'Sat 星期六', 'Sun 星期日'].map(k => {
     return k.replace(' ', '\n')
   })
}

设定固定长度换行, 用formatter自定义

 
xxxxxxxxxx
option.xAxis = {
   type : 'category',
   data : ['Mon星期一', 'Tue星期二', 'Wed星期三', 'Thu星期四', 'Fri星期五', 'Sat星期六', 'Sun星期日'],
   axisLabel: {
     rotate: 45,
     interval: 0,
     formatter: function (params) {
          if (!params) return ''
          var newParamsName = ''
          var paramsNameNumber = params.length
          var provideNumber = 3 //设置长度为3
          var rowNumber = Math.ceil(paramsNameNumber / provideNumber)
          if (paramsNameNumber > provideNumber) {
            for (var p = 0; p < rowNumber; p++) {
              var tempStr = ''
              var start = p * provideNumber
              var end = start + provideNumber
              if (p === rowNumber - 1) {
                tempStr = params.substring(start, paramsNameNumber)
              } else {
                tempStr = params.substring(start, end) + '\n'
              }
              newParamsName += tempStr
            }
          } else {
            newParamsName = params
          }
          return newParamsName
        }
   }
}

11. 图例的显示控制

legend.selectedMode 图例选择的模式，控制是否可以通过点击图例改变系列的显示状态。默认开启图例选择，可以设成 false 关闭。除此之外也可以设成 'single' 或者 'multiple' 使用单选或者多选模式。

legend的selected属性，图例选中状态表

 
xxxxxxxxxx
option.legend = {
  data: ['系列1', '系列2'],
  selected: {
    // 选中'系列1'
    '系列1': true,
    // 不选中'系列2'
    '系列2': false
  }
}
